  title1 'RADIX COMPANY';
  title2 'BIG VCR SALES';
  proc print data=vcr.sales;
     where unitsold>=30;
  run;
  title2 'BEST SALES';
  title3 'TOP MODELS';
  proc means max data=vcr.sales;
     where unitsold>=30;
     by type;
  run;
