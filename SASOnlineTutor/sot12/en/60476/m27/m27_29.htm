   infile orders end=last;
   retain Department Extension;
   input type $1. @;
   if type='D' then do;
      if _n_ > 1 then output;
      Total=0;
      input @3 department $10. @16 extension $5.;
   end;
   else if type='S' then do;
      input @16 Amount comma5.;
      total+amount;
      if last then output;
  end;
run;
