Managing Index Usage (continued)
Which program will write an informational message to the SAS log that indicates whether PROC SQL used an index in processing the query? 
proc sql msglevel=i;
   select itemid,description,qty
      from work.inventory
      where qty<=5;
 
proc sql msglevel=n;
   select itemid,description,qty
      from work.inventory
      where qty<=5;
 
options msglevel=i;
proc sql;
   select itemid,description,qty
      from work.inventory
      where qty<=5;
 
options msglevel=n;
proc sql;
   select itemid,description,qty
      from work.inventory
      where qty<=5;
