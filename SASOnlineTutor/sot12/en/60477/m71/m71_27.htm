Select the PROC SQL full join that will generate the identical output.
 

 
proc sql;
title 'Table Merged';
   select coalesce(names.id, ages.id)
          as ID, fname,  age
      from names full join ages
      on names.id = ages.id
      order by id;
 
proc sql;
title 'Table Merged';
   select coalesce(id)
          as ID, fname,  age
      from names full join ages
      on names.id = ages.id
      order by id;
 
proc sql;
title 'Table Merged';
   select names.id, fname,  age
      from names full join ages
      on names.id = ages.id
      order by id;
 
proc sql;
title 'Table Merged';
   select *
      from names full join ages
      on names.id = ages.id
      order by id;
